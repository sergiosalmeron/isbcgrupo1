110            /**
111             * Inits the OntoBridge with a DIG reasoner.
112             * Dig is a protocol (working over http) that allows conecting with any reasoner.
113             * @param reasonerHost Host where the reasoner is running.
114             * @param reasonerPort Port that the reasoner listens to.
115             */
116            public void initWithDIGReasoner(String reasonerHost, int reasonerPort)
117            {
118                    credits();
119                    //Create a default model
120                    Model cModel = ModelFactory.createDefaultModel();
121                    
122                    //Obtain the reasonerURL
123                    String reasonerURL = "http://" + reasonerHost + ":" + reasonerPort;
124                    
125                    //Create a reasoner configuration with the reasoner URL
126                    Resource conf = cModel.createResource();
127                    conf.addProperty(ReasonerVocabulary.EXT_REASONER_URL, cModel.createResource(reasonerURL));
128                    
129                    //Obtain the Dig reasoners Factory
130                    DIGReasonerFactory drf = (DIGReasonerFactory) ReasonerRegistry
131                                    .theRegistry().getFactory(DIGReasonerFactory.URI);
132    
133                    //Obtain a reasoner with our configuration
134                    DIGReasoner r = (DIGReasoner) drf.create(conf);
135    
136                    //Choose the Ontology Model Specification: OWL-DL in Memory
137                    OntModelSpec spec = new OntModelSpec(OntModelSpec.OWL_DL_MEM);
138                    spec.setReasoner(r);
139                    
140                    //Obtain the OntologyModel
141                    ONT_MODEL = ModelFactory.createOntologyModel(spec);
142            }

155            /**
156             * Enables or disables the inference. This method should be only used for performance reasons. 
157             * When disabling inference, current model is validated, backed-up and finally a new model is created without attached reasoner.
158             * If inference is enabled again the back-up model is restored.
159             * This method does not distinguish if the object was initializated without reasoner. If so, this method is unusefull.
160             * Any change performed in the ontology when inference is disabled will not be maintained when inference is enabled again, 
161             * because this method creates a copy of the current model.
162             * @param enabled Indicates if inference is enabled
163             */
164            public void setInference(boolean enabled)
165            {
166                    if(enabled)
167                    {
168                            if(backupModel != null)
169                            {
170                                    ONT_MODEL = backupModel;
171                                    backupModel = null;
172                            }
173                            else
174                                    org.apache.commons.logging.LogFactory.getLog(this.getClass()).warn("Inference already enabled.");
175                    }
176                    else
177                    {
178                            if(backupModel == null)
179                            {
180                                    //Validate the model (classify instances)
181                                    ONT_MODEL.validate();
182                                    //Backup model
183                                    backupModel = ONT_MODEL;
184                                    //Create a plain model with asserted and infered information (a-box + t-box)
185                                    Model plain = ModelFactory.createModelForGraph( ONT_MODEL.getGraph() );
186                                    //Create a new model without reasoner
187                                    ONT_MODEL = ModelFactory.createOntologyModel( OntModelSpec.OWL_MEM);
188                                    //Copy plain model into the model without reasoner
189                                    ONT_MODEL.add(plain);
190                                    //Copy ns prefixes
191                                    ONT_MODEL.setNsPrefixes(backupModel.getNsPrefixMap());
192                            }
193                            else
194                                    org.apache.commons.logging.LogFactory.getLog(this.getClass()).warn("Inference already disabled.");
195                    }
196            }

198            /**
199             * Loads an ontology using the initialized reasoner
200             * @param mainOntology Main ontology document
201             * @param subOntologies Collection of ontology documents used by the main ontology
202             * @param loadSubOntologies Indicates if the reasoner must load the subOntologies. 
203             *        If loaded, their content is included in the reasoning graph of the main ontology graph.
204             */
205            public void loadOntology(OntologyDocument mainOntology, Collection<OntologyDocument> subOntologies, boolean loadSubOntologies)
206            {
207                    //Configure local copies of the ontologies
208                    OntDocumentManager dm = ONT_MODEL.getDocumentManager();
209                    
210                    try {
211                            if(mainOntology.hasAltLocalFile())
212                                    dm.addAltEntry(mainOntology.getURL(), mainOntology.getLocalfile());
213                    } catch (NoLocalFileException e) {
214                            org.apache.commons.logging.LogFactory.getLog(this.getClass()).error(e);
215                    }
216                    
217                    for(OntologyDocument od : subOntologies)
218                    {
219                            try {
220                            if(od.hasAltLocalFile())
221                                    dm.addAltEntry(od.getURL(), od.getLocalfile());
222                            } catch (NoLocalFileException e) {
223                                    org.apache.commons.logging.LogFactory.getLog(this.getClass()).error(e);
224                            }
225                    }
226                    
227                    org.apache.commons.logging.LogFactory.getLog(this.getClass()).info("Loading Main Ontology: "+mainOntology.getURL());
228                    ONT_MODEL.read(mainOntology.getURL());  
229                    BASE_NS = (String)ONT_MODEL.getNsPrefixMap().get("");
230                    
231                    if(loadSubOntologies)
232                            for(OntologyDocument od : subOntologies)
233                            {
234                                    org.apache.commons.logging.LogFactory.getLog(this.getClass()).info("Loading Sub-Ontology: "+od.getURL());
235                                    ONT_MODEL.read(od.getURL());
236                            }
237                    
238                    
239                    org.apache.commons.logging.LogFactory.getLog(this.getClass()).info("Loading Complete");
240                    //Print Namespaces
241                    org.apache.commons.logging.LogFactory.getLog(this.getClass()).info( "Base Namespace: "+ BASE_NS );
242                    org.apache.commons.logging.LogFactory.getLog(this.getClass()).info( "Namespaces loaded: " + ONT_MODEL.getNsPrefixMap() );
243                    
244                    preCalculateProfs();
245                    
246            }

252            protected HashMap<String, Integer> profs = new HashMap<String,Integer>();
253            protected int maxDepth = 0;
254            
255            protected void preCalculateProfs()
256            {
257                    profs.clear();
258                    
259                    String thing = getThingURI();
260                    profs.put(thing, new Integer(0));
261                    
262                    HashSet<String> subclasses = new HashSet<String>();
263                    
264                    Iterator<String> iter = this.listRootClasses();
265                    boolean cont = true;
266                    int i = 1;
267                    while(cont)
268                    {
269                            subclasses  = new HashSet<String>();
270                            while(iter.hasNext())
271                            {
272                                    String _class = iter.next();
273                                    profs.put(_class, new Integer(i));
274                                            
275                                    Iterator<String> aux = this.listSubClasses(_class,true);
276                                    while(aux.hasNext())
277                                            subclasses.add(aux.next());     
278                            }
279                            cont = !subclasses.isEmpty();
280                            iter = subclasses.iterator();
281                            i++;
282                    }
283                    maxDepth = i-2;
284            }

287            {
288                    if(!_class.contains("#"))
289                            _class = getURI(_class);
290                    return profs.get(_class);

292            public int profInstance(String instance)
293            {
294                    int res = Integer.MAX_VALUE;
295                    for(Iterator<String> iter = listDeclaredBelongingClasses(instance); iter.hasNext();)
296                    {
297                            int prof = profConcept(iter.next());
298                            if(prof<res)
299                                    res = prof;
300                    }
301                    return res+1;
302            }

578            {
579                    ArrayList<String> list = new ArrayList<String>();
580                    OntClass ontC = obtainOntClass(ontClass);
581                    StmtIterator si = ONT_MODEL.getRawModel().listStatements(null, RDF.type, ontC);
582                    while(si.hasNext())
583                    {
584                            Statement st = si.nextStatement();
585                            list.add(st.getSubject().toString());
586                    }
587                    return list.iterator();
588            }

685            {
686                    ArrayList<String> list = new ArrayList<String>();
687                    OntClass ontR = obtainOntClass(ontClass);
688                    for(Iterator iter = ontR.listDeclaredProperties(); iter.hasNext();)
689                    {
690                            OntProperty prop = (OntProperty)iter.next();
691                            StmtIterator si = ONT_MODEL.listStatements(prop, RDFS.domain, ontR);
692                            if(si.hasNext())
693                                    list.add(prop.getURI());
694                    }
695                    return list.iterator();
696            }

733            /**
734             * Returns the properties with their corresponding values of an instance. The properties and values parameter are cleared.
735             * They are used to return the properties and values. property[i] has value values[i].
736             * @param instance source of the properties
737             * @param properties Array containing the name of the properties
738             * @param values Array containing the value of the property
739             */
740            public void listInstancePropertiesValues(String instance, List<String> properties, List<String> values)
741            {
742                    properties.clear();
743                    values.clear();
744                    
745                    
746                    OntResource ontI = obtainOntResource(instance);
747                    for(StmtIterator props = ontI.listProperties(); props.hasNext();)
748                    {
749                            Statement stmt = props.nextStatement();
750                            if(getShortName(stmt.getPredicate().toString()).equals("rdf:type"))
751                                    continue;
752                            properties.add(stmt.getPredicate().toString());
753                            values.add(stmt.getObject().toString());        
754                    }
755            }

1025             * Computes if the current ontology is valid. (Created by: Antonio Sanchez)

1028            public boolean isValid(boolean ignorewarnings) {

1031                            return vr.isValid();
1032                    else
1033                            return vr.isClean();

1036            /**
1037             * Computes if the current ontology is valid and returns a list of warnings and errors. (Created by: Antonio Sanchez)
1038             */
1039            public Iterator<String> validate() {
1040                    ValidityReport vr = ONT_MODEL.validate();
1041                    
1042                    Collection<String> reports = new ArrayList<String>();
1043                    
1044                    for(Iterator riter = vr.getReports(); riter.hasNext();)
1045                    {
1046                            ValidityReport.Report r = (ValidityReport.Report)riter.next();
1047                            String msg ="";
1048                            if(r.isError())
1049                                    msg += "[ERROR]";
1050                            else
1051                                    msg += "[WARNING]";
1052                            msg+="["+r.getType()+"]";
1053                            msg+=r.getDescription();
1054                            reports.add(msg);
1055                    }
1056                    
1057                    return reports.iterator();
1058            }

1079             * Returns the JENA OntClass referenced by the string.
1080             * This method checks if the string is a short or long URI.

1082            protected OntClass obtainOntClass(String ontClass)
1083            {
1084                    OntResource ontR = obtainOntResource(ontClass);
1085                    if(ontR != null)
1086                            if(ontR.canAs(OntClass.class))
1087                                    return ontR.asClass();
1088                    return null;    
1089            }

1092            /**
1093             * Returns the JENA OntProperty referenced by the string.
1094             * This method checks if the string is a short or long URI.
1095             */
1096            protected OntProperty obtainOntProperty(String ontProperty)
1097            {
1098                    OntResource ontR = obtainOntResource(ontProperty);
1099                    if(ontR != null)
1100                            if(ontR.isProperty())
1101                            {
1102                                    if(ontR.isDatatypeProperty())
1103                                            return ontR.asDatatypeProperty();
1104                                    return ontR.asProperty();
1105                            }
1106                    return null;    
1107            }
1108            
1109            /**
1110             * Returns the JENA OntResource referenced by the string.
1111             * This method checks if the string is a short or long URI.
1112             */
1113            protected OntResource obtainOntResource(String ontRes)
1114            {
1115                    OntResource ontR;
1116                    try {
1117                            ontR = ONT_MODEL.getOntResource(ontRes);
1118                            if(ontR != null)
1119                                    return ontR;
1120                    } catch (Exception e) {
1121                    }
1122                    
1123                    try {
1124                            String qname = ONT_MODEL.expandPrefix(ontRes);
1125                            ontR = ONT_MODEL.getOntResource(qname);
1126                            if(ontR != null)
1127                                    return ontR;
1128                    } catch (Exception e) {
1129                    }       
1130                    
1131                    try {
1132                            return ontR = ONT_MODEL.getOntResource(BASE_NS + ontRes);
1133                    } catch (Exception e) {
1134                    }
1135                    
1136                    org.apache.commons.logging.LogFactory.getLog(this.getClass()).error("Ontology Resource not found: "+ ontRes);
1137                    return null;
1138            }
